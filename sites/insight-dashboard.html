const emptyData = []; new Chart(document.getElementById("throughputChart"), { type: "line", data: { labels: emptyLabels, datasets: [{ label: "Throughput", data: emptyData }] } }); new Chart(document.getElementById("signalChart"), { type: "line", data: { labels: emptyLabels, datasets: [{ label: "Signal", data: emptyData }] } }); } // Call once on load initCharts(); // ======================= // LOAD FILE (START HERE) // ======================= function loadData() { const input = document.getElementById("fileInput"); if (!input.files.length) { alert("Please select a file first!"); return; } const file = input.files[0]; const fileName = file.name.toLowerCase(); // CSV / TXT if (fileName.endsWith(".csv") || fileName.endsWith(".txt")) { const reader = new FileReader(); reader.onload = function(e) { processTextData(e.target.result); }; reader.readAsText(file); } // Excel (future) else if (fileName.endsWith(".xlsx") || fileName.endsWith(".xls")) { alert("Excel support coming next 🔥"); } else { alert("Unsupported file format"); } } // ======================= // PROCESS TEXT DATA // ======================= function processTextData(text) { const rows = text.split("\n"); // HEADERS headers = rows[0].split(","); // DETECT INDUSTRY currentIndustry = detectIndustry(headers); console.log("Detected Industry:", currentIndustry); // DATA rawData = rows .slice(1) .filter(row => row.trim() !== "") .map(row => row.split(",")); console.log("DATA:", rawData); calculateKPI(); } // ======================= // DETECT INDUSTRY // ======================= function detectIndustry(headers) { const h = headers.join(" ").toLowerCase(); if ( h.includes("rsrp") || h.includes("rsrq") || h.includes("sinr") || h.includes("snir") || h.includes("throughput") || h.includes("drop") ) { return "telecom"; } if (h.includes("revenue") || h.includes("profit")) { return "finance"; } return "unknown"; } // ======================= // GET COLUMN INDEX // ======================= function getIndex(name) { return headers.findIndex(h => h.toLowerCase().includes(name.toLowerCase())); } // ======================= // KPI CALCULATION // ======================= function calculateKPI() { if (currentIndustry === "telecom") { let thrIndex = getIndex("throughput"); let dropIndex = getIndex("drop"); let totalThr = 0; let totalDrop = 0; let count = 0; rawData.forEach(row => { totalThr += Number(row[thrIndex]); totalDrop += Number(row[dropIndex]); count++; }); let avgThr = totalThr / count; let avgDrop = totalDrop / count; console.log("Avg Throughput:", avgThr); console.log("Avg Drop:", avgDrop); document.getElementById("throughput").innerText = avgThr.toFixed(0) + " Mbps"; } else { document.getElementById("throughput").innerText = "Unknown Data"; } }
